2. expressions:
	literal expression:
		<div class="value"> [literal] </div>
		i
		
	scope expression:
		<div class="scope">
			[define expression]*
			[expression]
		</div>
		article
		
	variable binding expression:
		<div class="define" id="[idenifier]"> [expression] </div>
		section
	
	variable use expression:
		<div class="variable"> [identifier] </div>
		a
	
	function binding expression:
		<div class="function" id="[identifier]?"> [expression] </div>
	
	argument expression:
		<div class="argument"></div>
		label
	
	function call expression:
		<div class="call">
			[function expression]
			[expression]
		</div>

		</div>

		</div>
